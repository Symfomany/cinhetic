<?php

namespace Ezap\ReceiptBundle\Form;

use Symfony\Component\Form\AbstractType;
use Symfony\Component\Form\FormBuilderInterface;
use Symfony\Component\OptionsResolver\OptionsResolverInterface;

class ReceiptType extends AbstractType
{
        /**
     * @param FormBuilderInterface $builder
     * @param array $options
     */
    public function buildForm(FormBuilderInterface $builder, array $options)
    {
        $builder
            ->add('paiement')
            ->add('datePurchase', "date", array('widget' => 'single_text','label' => "Date d'achat"))
            ->add('amountShow', null , array('label' => "Montant"))
            ->add('amountMade', null , array('label' => "Montant rendu"))
            ->add('store', null , array('label' => "Magasin"))
            ->add('zipcode', null , array('label' => "Code postal"))
            ->add('city', null , array('label' => "Ville"))
            ->add('tel', null , array('label' => "Téléphone"))
            ->add('tva')
            ->add('loyalty', null , array('label' => "Fidelité"))
            ->add('nbArticles', null , array('label' => "Nombre d'articles"))
            ->add('url')
            ->add('dateReceipt', 'date', array('widget' => 'single_text', 'label' => "Date de validité"))
            ->add('refTransaction')
            ->add('message', null, array('attr' => array('class' => 'ckeditor')))
        ;
    }
    
    /**
     * @param OptionsResolverInterface $resolver
     */
    public function setDefaultOptions(OptionsResolverInterface $resolver)
    {
        $resolver->setDefaults(array(
            'data_class' => 'Ezap\ReceiptBundle\Entity\Receipt'
        ));
    }

    /**
     * @return string
     */
    public function getName()
    {
        return 'ezap_receiptbundle_receipt';
    }
}
