<?php

namespace Ezap\ReceiptBundle\Entity;

use Doctrine\ORM\Mapping as ORM;
use Ezap\PublicBundle\Entity\User;

/**
 * Receipt
 *
 * @ORM\Table(name="receipt")
 * @ORM\Entity(repositoryClass="Ezap\ReceiptBundle\Repository\ReceiptRepository")
 * */
class Receipt
{

    /**
     * Constructor of Receipt
     */
    public function __construct(){
        $this->dateCreated = new \Datetime('now');
    }

    /**
     * @var integer
     *
     * @ORM\Column(name="id", type="integer", nullable=false)
     * @ORM\Id
     * @ORM\GeneratedValue(strategy="IDENTITY")
     */
    private $id;

    /**
     * @var integer
     * @ORM\Column(name="id_user", type="integer", nullable=true)
     * @ORM\OneToMany(targetEntity="EzapPublicBundle\User", mappedBy="receipts")
     */
    private $user;

    /**
     * @var string
     *
     * @ORM\Column(name="message", type="text", nullable=true)
     */
    private $message;

    /**
     * @var integer
     *
     * @ORM\Column(name="paiement", type="string", nullable=true)
     */
    private $paiement;

    /**
     * @var \DateTime
     *
     * @ORM\Column(name="date_purchase", type="datetime", nullable=true)
     */
    private $datePurchase;

    /**
     * @var integer
     *
     * @ORM\Column(name="amount_show", type="float", nullable=true)
     */
    private $amountShow;

    /**
     * @var integer
     *
     * @ORM\Column(name="amount_made", type="float", nullable=true)
     */
    private $amountMade;

    /**
     * @var integer
     *
     * @ORM\Column(name="code_toilet", type="integer", nullable=true)
     */
    private $codeToilet;

    /**
     * @var string
     *
     * @ORM\Column(name="store", type="string", length=200, nullable=true)
     */
    private $store;


    /**
     * @var string
     *
     * @ORM\Column(name="state", type="boolean")
     */
    private $state;

    /**
     * @var integer
     *
     * @ORM\Column(name="zipcode", type="integer", nullable=true)
     */
    private $zipcode;

    /**
     * @var string
     *
     * @ORM\Column(name="city", type="string", length=200, nullable=true)
     */
    private $city;

    /**
     * @var string
     *
     * @ORM\Column(name="tel", type="string", length=200, nullable=true)
     */
    private $tel;

    /**
     * @var string
     *
     * @ORM\Column(name="tva", type="string", length=200, nullable=true)
     */
    private $tva;

    /**
     * @var string
     *
     * @ORM\Column(name="loyalty", type="string", length=200, nullable=true)
     */
    private $loyalty;

    /**
     * @var integer
     *
     * @ORM\Column(name="nb_articles", type="integer", nullable=true)
     */
    private $nbArticles;

    /**
     * @var string
     *
     * @ORM\Column(name="url", type="string", length=400, nullable=true)
     */
    private $url;

    /**
     * @var string
     *
     * @ORM\Column(name="date_receipt", type="datetime", length=300, nullable=true)
     */
    private $dateReceipt;

    /**
     * @var integer
     *
     * @ORM\Column(name="ref_transaction", type="integer", nullable=true)
     */
    private $refTransaction;

    /**
     * @var \DateTime
     *
     * @ORM\Column(name="date_created", type="datetime", nullable=true)
     */
    private $dateCreated;

    /**
     * @var \DateTime
     *
     * @ORM\Column(name="rtf", type="text", nullable=true)
     */
    private $rtf;


    /**
     *
     * @ORM\Column(name="longitude", type="float", nullable=true)
     */
    private $longitude;

    /**
     *
     * @ORM\Column(name="latitude", type="float", nullable=true)
     */
    private $latitude;



    /**
     * Get id
     *
     * @return integer 
     */
    public function getId()
    {
        return $this->id;
    }


    /**
     * Set message
     *
     * @param string $message
     * @return Receipt
     */
    public function setMessage($message)
    {
        $this->message = $message;

        return $this;
    }

    /**
     * Get message
     *
     * @return string 
     */
    public function getMessage()
    {
        return $this->message;
    }

    /**
     * Set paiement
     *
     * @param integer $paiement
     * @return Receipt
     */
    public function setPaiement($paiement)
    {
        $this->paiement = $paiement;

        return $this;
    }

    /**
     * Get paiement
     *
     * @return integer 
     */
    public function getPaiement()
    {
        return $this->paiement;
    }

    /**
     * Set datePurchase
     *
     * @param \DateTime $datePurchase
     * @return Receipt
     */
    public function setDatePurchase($datePurchase)
    {
        $this->datePurchase = $datePurchase;

        return $this;
    }

    /**
     * Get datePurchase
     *
     * @return \DateTime 
     */
    public function getDatePurchase()
    {
        return $this->datePurchase;
    }

    /**
     * Set amountShow
     *
     * @param integer $amountShow
     * @return Receipt
     */
    public function setAmountShow($amountShow)
    {
        $this->amountShow = $amountShow;

        return $this;
    }

    /**
     * Get amountShow
     *
     * @return integer 
     */
    public function getAmountShow()
    {
        return $this->amountShow;
    }

    /**
     * Set amountMade
     *
     * @param integer $amountMade
     * @return Receipt
     */
    public function setAmountMade($amountMade)
    {
        $this->amountMade = $amountMade;

        return $this;
    }

    /**
     * Get amountMade
     *
     * @return integer 
     */
    public function getAmountMade()
    {
        return $this->amountMade;
    }

    /**
     * Set codeToilet
     *
     * @param integer $codeToilet
     * @return Receipt
     */
    public function setCodeToilet($codeToilet)
    {
        $this->codeToilet = $codeToilet;

        return $this;
    }

    /**
     * Get codeToilet
     *
     * @return integer 
     */
    public function getCodeToilet()
    {
        return $this->codeToilet;
    }

    /**
     * Set store
     *
     * @param string $store
     * @return Receipt
     */
    public function setStore($store)
    {
        $this->store = $store;

        return $this;
    }

    /**
     * Get store
     *
     * @return string 
     */
    public function getStore()
    {
        return $this->store;
    }

    /**
     * Set zipcode
     *
     * @param integer $zipcode
     * @return Receipt
     */
    public function setZipcode($zipcode)
    {
        $this->zipcode = $zipcode;

        return $this;
    }

    /**
     * Get zipcode
     *
     * @return integer 
     */
    public function getZipcode()
    {
        return $this->zipcode;
    }

    /**
     * Set city
     *
     * @param string $city
     * @return Receipt
     */
    public function setCity($city)
    {
        $this->city = $city;

        return $this;
    }

    /**
     * Get city
     *
     * @return string 
     */
    public function getCity()
    {
        return $this->city;
    }

    /**
     * Set tel
     *
     * @param string $tel
     * @return Receipt
     */
    public function setTel($tel)
    {
        $this->tel = $tel;

        return $this;
    }

    /**
     * Get tel
     *
     * @return string 
     */
    public function getTel()
    {
        return $this->tel;
    }

    /**
     * Set tva
     *
     * @param string $tva
     * @return Receipt
     */
    public function setTva($tva)
    {
        $this->tva = $tva;

        return $this;
    }

    /**
     * Get tva
     *
     * @return string 
     */
    public function getTva()
    {
        return $this->tva;
    }

    /**
     * Set loyalty
     *
     * @param string $loyalty
     * @return Receipt
     */
    public function setLoyalty($loyalty)
    {
        $this->loyalty = $loyalty;

        return $this;
    }

    /**
     * Get loyalty
     *
     * @return string 
     */
    public function getLoyalty()
    {
        return $this->loyalty;
    }

    /**
     * Set nbArticles
     *
     * @param integer $nbArticles
     * @return Receipt
     */
    public function setNbArticles($nbArticles)
    {
        $this->nbArticles = $nbArticles;

        return $this;
    }

    /**
     * Get nbArticles
     *
     * @return integer 
     */
    public function getNbArticles()
    {
        return $this->nbArticles;
    }

    /**
     * Set url
     *
     * @param string $url
     * @return Receipt
     */
    public function setUrl($url)
    {
        $this->url = $url;

        return $this;
    }

    /**
     * Get url
     *
     * @return string 
     */
    public function getUrl()
    {
        return $this->url;
    }

    /**
     * Set dateReceipt
     *
     * @param string $dateReceipt
     * @return Receipt
     */
    public function setDateReceipt($dateReceipt)
    {
        $this->dateReceipt = $dateReceipt;

        return $this;
    }

    /**
     * Get dateReceipt
     *
     * @return string 
     */
    public function getDateReceipt()
    {
        return $this->dateReceipt;
    }

    /**
     * Set refTransaction
     *
     * @param integer $refTransaction
     * @return Receipt
     */
    public function setRefTransaction($refTransaction)
    {
        $this->refTransaction = $refTransaction;

        return $this;
    }

    /**
     * Get refTransaction
     *
     * @return integer 
     */
    public function getRefTransaction()
    {
        return $this->refTransaction;
    }

    /**
     * Set dateCreated
     *
     * @param \DateTime $dateCreated
     * @return Receipt
     */
    public function setDateCreated($dateCreated)
    {
        $this->dateCreated = $dateCreated;

        return $this;
    }

    /**
     * Get dateCreated
     *
     * @return \DateTime 
     */
    public function getDateCreated()
    {
        return $this->dateCreated;
    }

    /**
     * Set state
     *
     * @param boolean $state
     * @return Receipt
     */
    public function setState($state)
    {
        $this->state = $state;

        return $this;
    }

    /**
     * Get state
     *
     * @return boolean 
     */
    public function getState()
    {
        return $this->state;
    }

    /**
     * Set rtf
     *
     * @param \tet $rtf
     * @return Receipt
     */
    public function setRtf(\tet $rtf)
    {
        $this->rtf = $rtf;

        return $this;
    }

    /**
     * Get rtf
     *
     * @return \tet 
     */
    public function getRtf()
    {
        return $this->rtf;
    }


    /**
     * Set user
     *
     * @param integer $user
     * @return Receipt
     */
    public function setUser(User $user)
    {
        $this->user = $user;

        return $this;
    }

    /**
     * Get user
     *
     * @return integer 
     */
    public function getUser()
    {
        return $this->user;
    }

    /**
     * Set longitude
     *
     * @param float $longitude
     * @return Receipt
     */
    public function setLongitude($longitude)
    {
        $this->longitude = $longitude;

        return $this;
    }

    /**
     * Get longitude
     *
     * @return float 
     */
    public function getLongitude()
    {
        return $this->longitude;
    }

    /**
     * Set latitude
     *
     * @param float $latitude
     * @return Receipt
     */
    public function setLatitude($latitude)
    {
        $this->latitude = $latitude;

        return $this;
    }

    /**
     * Get latitude
     *
     * @return float 
     */
    public function getLatitude()
    {
        return $this->latitude;
    }
}
