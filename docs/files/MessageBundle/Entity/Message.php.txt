<?php

namespace Ezap\MessageBundle\Entity;

use Doctrine\ORM\Mapping as ORM;
use Ezap\PublicBundle\Entity\User;

/**
 * Receipt
 * @ORM\Table(name="message")
 * @ORM\Entity(repositoryClass="Ezap\MessageBundle\Repository\MessageRepository")
 * */
class Message
{

    /**
     * Constructor of Receipt
     */
    public function __construct(){
        $this->dateCreated =  new \Datetime('now');
        $this->state =  1;
    }

    /**
     * @var integer
     *
     * @ORM\Column(name="id", type="integer", nullable=false)
     * @ORM\Id
     * @ORM\GeneratedValue(strategy="IDENTITY")
     */
    private $id;

    /**
     * @var integer
     * @ORM\Column(name="user_id", type="integer", nullable=true)
     * @ORM\OneToMany(targetEntity="EzapPublicBundle\User", mappedBy="messages")
     */
    private $user;

    /**
     * @var integer
     * @ORM\Column(name="recepit_id", type="integer", nullable=true)
     * @ORM\OneToMany(targetEntity="EzapReceiptBundle\Receipt", mappedBy="messages")
     */
    private $receipt;

    /**
     * @var string
     *
     * @ORM\Column(name="message", type="text", nullable=true)
     */
    private $message;

    /**
     * @var string
     *
     * @ORM\Column(name="state", type="integer", nullable=true)
     */
    private $state;

    /**
     * @var \DateTime
     *
     * @ORM\Column(name="date_created", type="datetime", nullable=true)
     */
    private $dateCreated;


    /**
     * Get id
     *
     * @return integer 
     */
    public function getId()
    {
        return $this->id;
    }

    /**
     * Set user
     *
     * @param integer $user
     * @return Message
     */
    public function setUser($user)
    {
        $this->user = $user;

        return $this;
    }

    /**
     * Get user
     *
     * @return integer 
     */
    public function getUser()
    {
        return $this->user;
    }

    /**
     * Set receipt
     *
     * @param integer $receipt
     * @return Message
     */
    public function setReceipt($receipt)
    {
        $this->receipt = $receipt;

        return $this;
    }

    /**
     * Get receipt
     *
     * @return integer 
     */
    public function getReceipt()
    {
        return $this->receipt;
    }

    /**
     * Set message
     *
     * @param string $message
     * @return Message
     */
    public function setMessage($message)
    {
        $this->message = $message;

        return $this;
    }

    /**
     * Get message
     *
     * @return string 
     */
    public function getMessage()
    {
        return $this->message;
    }

    /**
     * Set dateCreated
     *
     * @param \DateTime $dateCreated
     * @return Message
     */
    public function setDateCreated($dateCreated)
    {
        $this->dateCreated = $dateCreated;

        return $this;
    }

    /**
     * Get dateCreated
     *
     * @return \DateTime 
     */
    public function getDateCreated()
    {
        return $this->dateCreated;
    }

    /**
     * Set state
     *
     * @param integer $state
     * @return Message
     */
    public function setState($state)
    {
        $this->state = $state;

        return $this;
    }

    /**
     * Get state
     *
     * @return integer 
     */
    public function getState()
    {
        return $this->state;
    }
}
